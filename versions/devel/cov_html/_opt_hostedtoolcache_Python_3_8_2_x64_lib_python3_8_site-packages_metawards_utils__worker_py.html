15@contextmanager 

16def silence_output(): 

17 """Nice way to silence stdout and stderr - thanks to 

18 Emil Stenström in 

19 https://stackoverflow.com/questions/6735917/redirecting-stdout-to-nothing-in-python 

20 """ 

21 new_out = open(os.devnull, "w") 

22 old_out = sys.stdout 

23 sys.stdout = new_out 

24 

25 new_err = open(os.devnull, "w") 

26 old_err = sys.stderr 

27 sys.stderr = new_err 

28 

29 try: 

30 yield new_out 

31 finally: 

32 sys.stdout = old_out 

33 sys.stderr = old_err

36def prepare_worker(params: Parameters, options): 

37 """Prepare a worker to receive work to run a model using the passed 

38 parameters. This will build the network specified by the 

39 parameters and will store it in global memory ready to 

40 be used for a model run. Note that these are 

41 silent, printing nothing to stdout or stderr 

42 """ 

43 # switch off printing to stdout and stderr 

44 with silence_output(): 

45 global global_network 

46 

47 max_nodes = options["max_nodes"] 

48 max_links = options["max_links"] 

49 

50 del options["max_nodes"] 

51 del options["max_links"] 

52 

53 if global_network is None: 

54 global_network = Network.build(params=params, 

55 calculate_distances=True, 

56 profile=False, 

57 max_nodes=max_nodes, 

58 max_links=max_links) 

59 

60 else: 

61 global_network.update(params)

64def run_worker(arguments): 

65 """Ask the worker to run a model using the passed variables and 

66 options. This will write to options['output_dir'] and will 

67 also return the population object that contains the final 

68 population data. 

69 

70 WARNING - the iterator and extractor arguments rely on the 

71 workers starting in the same directory as the main process, 

72 so that they can load the same python files (if the user 

73 is using a custom iterator or extractor) 

74 """ 

75 params = arguments["params"] 

76 options = arguments["options"] 

77 

78 # first, build and prepare the network 

79 prepare_worker(params, options) 

80 

81 # next, run the job, writing to output 

82 from ._runner import redirect_output 

83 

84 outdir = options["output_dir"] 

85 auto_bzip = options["auto_bzip"] 

86 del options["auto_bzip"] 

87 

88 # if the user wanted to remove this directory then they would 

89 # have done so in the main process - no need to check again 

90 with OutputFiles(outdir, check_empty=False, force_empty=False, 

91 prompt=None, auto_bzip=auto_bzip) as output_dir: 

92 options["output_dir"] = output_dir 

93 

94 with redirect_output(output_dir.get_path()): 

95 global global_network 

96 output = global_network.run(**options) 

97 

98 return output
